# indexOf

// indexOf
function (arr) {
    let array = [];
    for( var i = 0; i < arr.length; i++ ){
        if(array.indexOf(arr[i]) === -1){
            array.push(arr[i]);
        };
    };

# sort

// sort排序
function (arr) {
    arr.sort();
    let array = [arr[0]]
    for( var i = 1; i < arr.length; i++ ){
        if(arr[i] !== arr[i-1]){
            array.push(arr[i])
        }
    }
}

# includes

// includes
function (arr) {
    let array = [];
    for( var i = 0; i < arr.length; i++ ){
        if(!array.includes(arr[i])){
            array.push(arr[i])
        }
    }
}

# reduce

// reduce
function (arr) {
    arr.reduce((prev,cur) => prev.includes(cur) ? prev : [...prev,cur],[]);
}
