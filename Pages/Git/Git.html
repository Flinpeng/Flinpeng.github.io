# Git 命令

// 查看状态
git status
// 添加所有修改
git add .
// 提交修改
git commit -m  ""
// 拉取
git pull
// 推送 
git push
// 切换分支
git checkout name
// 创建新分支
git branch name
// copy并创建切换到新分支
git checkout -b name
// 查看本地分支
git branch
// 查看所有本地和远程分支
git branch -a 
// 把新建的分支push到远端
git push origin name 
// 删除本地分支
git branch -d name
// 删除远程分支
git push origin --delete name
// 查看本地分支和远程分支的关联关系
git branch -vv
// 设置本地和远程分支的联系
git branch --set-upstream-to=origin/远程分支的名字 本地分支的名字
