# vue官方文档里介绍可以返回一个会被合并的对象

vue官方文档介绍

const TerserPlugin = require('terser-webpack-plugin')
module.exports = {
  configureWebpack: (config)=>{
    if(process.env.NODE_ENV === 'production'){
      // 返回一个将会被合并的对象
      return {
        optimization:{
  		  minimizer: [
			new TerserPlugin({
				sourceMap:false,
				terserOptions:{
				  compress:{
				    drop_console : true
				  }
				}
			})
          ]
		}
	  }
    }
  }
}

# axios 并发请求

//全局挂载axios
import axios from 'axios'
Vue.prototype.$axios = axios
// 请求方法（已封装请求）,数组中为需要请求的接口
this.$axios.all([getInfo(),getPrice()]).then(
  // spread函数在所有的请求执行完成后执行，spread()函数的返回值和请求的顺序是一致的
  this.$axios.spread((info,price)=>{
    console.log('所有请求完成')
    console.log('请求1结果',info)
    console.log('请求2结果',price)
  })
)

  return {
