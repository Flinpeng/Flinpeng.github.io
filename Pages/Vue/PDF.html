# PDF文件渲染组件

TIP

在vue中引入vue-pdf插件来进行处理

npm install vue-pdf --save
1
<template>
    <div class="pdfBox">
        <pdf
            class="pdf"
            v-if="pdfSrc"
            :src="pdfSrc"
            :page="currentPage"
            @num-pages="pageCount=$event"
            @page-loaded="currentPage=$event"
            @loaded="loadPdfHandler"
        ></pdf>
        <div class="btn" v-if="pdfSrc">
            <button
                @click="changePdfPage(0)"
                :style="{'background':background,'color':btnColor}"
            >上一页</button>
            <h3 :style="{'color':fontColor}">{{currentPage}} / {{pageCount}}</h3>
            <button
                @click="changePdfPage(1)"
                :style="{'background':background,'color':btnColor}"
            >下一页</button>
        </div>
    </div>
</template>


<script>
import pdf from "vue-pdf";
export default {
    name: "Pdf",
    components: {
        pdf,
    },
    props: {
        pdfUrl: {
            type: String,
        },
        background: {
            type: String,
            default: "#ff5400",
        },
        btnColor: {
            type: String,
            default: "#fff",
        },
        fontColor: {
            type: String,
            default: "#ff5400",
        },
    },
    data() {
        return {
            pdfSrc: "",
            currentPage: 0, // pdf文件页码
            pageCount: 0, // pdf文件总页数
        };
    },
    mounted() {
        this.pdfSrc = this.pdfUrl;
    },
    methods: {
        // pdf加载
        loadPdfHandler(e) {
            this.currentPage = 1; // 加载的时候先加载第一页
        },
        // 改变PDF页码,val传过来区分上一页下一页的值,0上一页,1下一页
        changePdfPage(val) {
            if (val === 0 && this.currentPage > 1) {
                this.currentPage--;
            }
            if (val === 1 && this.currentPage < this.pageCount) {
                this.currentPage++;
            }
        },
    },
};
</script>
